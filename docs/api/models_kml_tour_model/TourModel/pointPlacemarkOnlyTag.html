String pointPlacemarkOnlyTag(
    String placemarkName, double lat, double long, String styleID) {
  String content = '';
  //  <styleUrl>#placemark-$placemarkIndex-style</styleUrl>
  content += '''
<Placemark>
    <name>$placemarkName</name>
    <styleUrl>#$styleID</styleUrl>

    </Point>
  </Placemark>
''';
